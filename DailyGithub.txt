for i in {1..20}
do
  echo "Commit $i on $(date)" >> daily_log.txt
  git add daily_log.txt
  git commit -m "Daily commit $i: $(date +"%Y-%m-%d %H:%M:%S")"
  sleep 2  # Wait 2 seconds
done

git push origin main
